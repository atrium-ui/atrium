## Installation

```bash
npm i @sv/scroll-lock
```

## Usage

```ts
import { ScrollLock } from '@sv/scroll-lock';

const scrollLock = new ScrollLock();

// lock page scrolling
scrollLock.enable();
// unlock page scrolling
scrollLock.disable();
```
